135     @Subscribe
136     @AllowConcurrentEvents
137     public void onPageCopyEvent(PageCopyEvent pageCopyEvent) {
138         if (pageCopyEvent.getException() != null) {
139             return;
140         }
141 
142         final PageCopyContext pageCopyContext = pageCopyEvent.getPageCopyContext();
143         final HstRequestContext requestContext = pageCopyContext.getRequestContext();
144         final Mount sourceMount = pageCopyContext.getEditingMount();
145         final Mount targetMount = pageCopyContext.getTargetMount();
146 
147         final String sourceContentBasePath = sourceMount.getContentPath().intern();
148         final String targetContentBasePath = targetMount.getContentPath().intern();
149 
150         // synchronize interned targetContentBasePath to disallow concurrent document copying on the same target channel
151         synchronized (targetContentBasePath) {
152             try {
153                 onBeforePageCopyEvent(pageCopyEvent);
154 
155                 final Node sourceContentBaseNode = requestContext.getSession().getNode(sourceContentBasePath);
156                 final Node targetContentBaseNode = requestContext.getSession().getNode(targetContentBasePath);
157                 String sourceTranslationLanguage = HippoFolderDocumentUtils
158                         .getHippoTranslationLanguage(sourceContentBaseNode);
159                 String targetTranslationLanguage = HippoFolderDocumentUtils
160                         .getHippoTranslationLanguage(targetContentBaseNode);
161 
162                 if (StringUtils.isBlank(sourceTranslationLanguage)) {
163                     throw new IllegalStateException("Blank translation language in the source base content at '"
164                             + sourceContentBasePath + "'.");
165                 }
166 
167                 if (StringUtils.isBlank(targetTranslationLanguage)) {
168                     throw new IllegalStateException("Blank translation language in the target base content at '"
169                             + targetContentBasePath + "'.");
170                 }
171 
172                 if (isCopyDocumentsLinkedBySourcePage()) {
173                     if (StringUtils.equals(sourceContentBasePath, targetContentBasePath)) {
174                         log.info(
175                                 "No need to copy documents because the source and target channel have the same content base path: {}'",
176                                 sourceContentBasePath);
177                     } else {
178                         if (StringUtils.equals(sourceTranslationLanguage, targetTranslationLanguage)) {
179                             throw new IllegalStateException(
180                                     "The same translation language of the source and the target base content. Source='"
181                                             + sourceContentBasePath + "'. Target='" + targetContentBasePath + "'.");
182                         }
183 
184                         final Set<String> documentPathSet = getDocumentPathSetInPage(pageCopyContext);
185 
186                         if (!documentPathSet.isEmpty()) {
187                             if (!StringUtils.equals(sourceMount.getContentPath(), targetMount.getContentPath())) {
188                                 copyDocuments(pageCopyContext.getRequestContext().getSession(), documentPathSet,
189                                         sourceContentBaseNode, targetContentBaseNode);
190                             } else {
191                                 log.info(
192                                         "Linked document copying step skipped because the content path of the target mount is the same as that of the source mount.");
193                             }
194                         } else {
195                             log.info("No linked document founds in the source page.");
196                         }
197                     }
198                 } else {
199                     log.info(
200                             "Linked document copying step skipped because 'copyDocumentsLinkedBySourcePage' is turned off.");
201                 }
202 
203                 updateTargetHstConfiguration(pageCopyContext);
204 
205                 onAfterPageCopyEvent(pageCopyEvent);
206             } catch (ClientException e) {
207                 log.error("Failed to handle page copy event properly.", e);
208                 pageCopyEvent.setException(e);
209             } catch (Exception e) {
210                 log.error("Failed to handle page copy event properly.", e);
211                 final String clientMessage = "Failed to handle page copy event properly. " + e.toString();
212                 pageCopyEvent.setException(new ClientException(clientMessage, ClientError.ITEM_CANNOT_BE_CLONED,
213                         Collections.singletonMap("errorReason", clientMessage)));
214             }
215         }
216     }

228     private void copyDocuments(final Session session, final Set<String> sourceDocumentPathSet,
229             final Node sourceContentBaseNode, final Node targetContentBaseNode) {
230         try {
231             final String sourceContentBasePath = sourceContentBaseNode.getPath();
232             final String targetContentBasePath = targetContentBaseNode.getPath();
233             final String targetTranslationLanguage = HippoFolderDocumentUtils
234                     .getHippoTranslationLanguage(targetContentBaseNode);
235 
236             Node sourceDocumentHandleNode;
237             Node targetDocumentHandleNode;
238             String targetDocumentAbsPath;
239             String targetFolderAbsPath;
240             String targetFolderRelPath;
241 
242             for (String sourceDocumentPath : sourceDocumentPathSet) {
243                 if (StringUtils.startsWith(sourceDocumentPath, "/")) {
244                     log.info(
245                             "Skipping '{}' because it's an absolute jcr path, not relative to source mount content base",
246                             sourceDocumentPath);
247                     continue;
248                 }
249 
250                 if (!sourceContentBaseNode.hasNode(sourceDocumentPath)) {
251                     log.info("Skipping '{}' because it doesn't exist under '{}'.", sourceDocumentPath, sourceContentBasePath);
252                     continue;
253                 }
254 
255                 sourceDocumentHandleNode = HippoFolderDocumentUtils
256                         .getHippoDocumentHandle(sourceContentBaseNode.getNode(sourceDocumentPath));
257 
258                 if (sourceDocumentHandleNode == null) {
259                     log.info("Skipping '{}' because there's no document at the location under '{}'.", sourceDocumentPath,
260                             sourceContentBasePath);
261                     continue;
262                 }
263 
264                 targetDocumentHandleNode = findTargetTranslatedDocumentHandleNode(targetContentBaseNode, sourceDocumentHandleNode);
265 
266                 if (targetDocumentHandleNode != null) {
267                     log.info("Skipping '{}' because there exists a translated document at '{}'.", sourceDocumentPath,
268                             targetDocumentHandleNode.getPath());
269                     continue;
270                 }
271 
272                 targetDocumentAbsPath = resolveTargetDocumentAbsPath(sourceContentBaseNode, targetContentBaseNode, sourceDocumentPath);
273 
274                 if (HippoFolderDocumentUtils.documentExists(session, targetDocumentAbsPath)) {
275                     log.info("Skipping '{}' because it already exists under '{}'.", sourceDocumentPath, targetContentBasePath);
276                     continue;
277                 }
278 
279                 targetFolderAbsPath = StringUtils.substringBeforeLast(targetDocumentAbsPath, "/");
280                 targetFolderRelPath = StringUtils.substringAfter(targetFolderAbsPath, targetContentBasePath + "/");
281 
282                 if (HippoFolderDocumentUtils.folderExists(session, targetFolderAbsPath)) {
283                     Node targetFolderNode = session.getNode(targetFolderAbsPath);
284 
285                     if (!targetFolderNode.isNodeType(HippoTranslationNodeType.NT_TRANSLATED)) {
286                         final String clientMessage = "Cannot copy documents because the target folder at '"
287                                 + targetFolderAbsPath + "' is not type of " + HippoTranslationNodeType.NT_TRANSLATED
288                                 + ".";
289                         throw new ClientException(clientMessage, ClientError.INVALID_NODE_TYPE,
290                                 Collections.singletonMap("errorReason", clientMessage));
291                     } else {
292                         Node sourceFolderNode = sourceDocumentHandleNode.getParent();
293                         String sourceFolderTranslationId = JcrUtils.getStringProperty(sourceFolderNode,
294                                 HippoTranslationNodeType.ID, null);
295                         String targetFolderTranslationId = JcrUtils.getStringProperty(targetFolderNode,
296                                 HippoTranslationNodeType.ID, null);
297                         if (!StringUtils.equals(sourceFolderTranslationId, targetFolderTranslationId)) {
298                             final String clientMessage = "Cannot copy documents because the translation ID of target folder at '"
299                                     + targetFolderAbsPath + "' doesn't match with that of source folder at '"
300                                     + sourceFolderNode.getPath() + "'. '" + targetFolderTranslationId
301                                     + "' (target) vs. '" + sourceFolderTranslationId + "' (source).";
302                             throw new ClientException(clientMessage, ClientError.ITEM_CANNOT_BE_CLONED,
303                                     Collections.singletonMap("errorReason", clientMessage));
304                         }
305                     }
306                 } else {
307                     String sourceFolderRelPath = sourceDocumentHandleNode.getParent().getPath()
308                             .substring(sourceContentBasePath.length() + 1);
309 
310                     translateFolders(session, sourceContentBaseNode, sourceFolderRelPath, targetContentBaseNode,
311                             targetFolderRelPath, targetTranslationLanguage);
312                 }
313 
314                 getDocumentManagementServiceClient().translateDocument(sourceDocumentHandleNode.getPath(),
315                         targetTranslationLanguage, sourceDocumentHandleNode.getName());
316             }
317         } catch (ClientException e) {
318             throw e;
319         } catch (Exception e) {
320             final String clientMessage = "Failed to copy all the linked documents. " + e.toString();
321             throw new ClientException(clientMessage, ClientError.ITEM_CANNOT_BE_CLONED,
322                     Collections.singletonMap("errorReason", clientMessage));
323         }
324     }

326     /**
327      * Resolves target document absolute path under {@code targetContentBaseNode},
328      * corresponding to the {@code sourceDocumentPath} under {@code sourceContentBaseNode}.
329      * @param sourceContentBaseNode source content base folder node
330      * @param targetContentBaseNode target content base folder node
331      * @param sourceDocumentPath source document relative path
332      * @return corresponding target document absolute path
333      * @throws RepositoryException if any repository exception occurs
334      */
335     private String resolveTargetDocumentAbsPath(final Node sourceContentBaseNode, final Node targetContentBaseNode,
336             final String sourceDocumentPath) throws RepositoryException {
337         Node sourceDocumentHandleNode = sourceContentBaseNode.getNode(sourceDocumentPath);
338         Node sourceFolderNode = sourceDocumentHandleNode.getParent();
339         Node targetFolderNode = findTargetTranslatedFolderNode(targetContentBaseNode, sourceFolderNode);
340 
341         if (targetFolderNode != null) {
342             return targetFolderNode.getPath() + "/" + sourceDocumentHandleNode.getName();
343         }
344 
345         Stack<String> targetFolderNameStack = new Stack<>();
346         targetFolderNameStack.push(sourceFolderNode.getName());
347 
348         sourceFolderNode = sourceFolderNode.getParent();
349 
350         while (!sourceFolderNode.isSame(sourceContentBaseNode)) {
351             targetFolderNode = findTargetTranslatedFolderNode(targetContentBaseNode, sourceFolderNode);
352 
353             if (targetFolderNode != null) {
354                 String folderPath = StringUtils.removeStart(targetFolderNode.getPath(),
355                         targetContentBaseNode.getPath() + "/");
356                 targetFolderNameStack.push(folderPath);
357                 break;
358             } else {
359                 targetFolderNameStack.push(sourceFolderNode.getName());
360             }
361 
362             sourceFolderNode = sourceFolderNode.getParent();
363         }
364 
365         return targetContentBaseNode.getPath() + "/" + StringUtils.join(popAllToList(targetFolderNameStack), "/") + "/"
366                 + sourceDocumentHandleNode.getName();
367     }

369     /**
370      * Find translated folder node under {@code targetContentBaseNode} for the {@code sourceFolderNode}.
371      * @param targetContentBaseNode target content base folder node
372      * @param sourceFolderNode source folder node
373      * @return translated folder node under {@code targetContentBaseNode} for the {@code sourceFolderNode}
374      * @throws RepositoryException if repository exception occurs
375      */
376     private Node findTargetTranslatedFolderNode(final Node targetContentBaseNode, final Node sourceFolderNode)
377             throws RepositoryException {
378         if (!sourceFolderNode.isNodeType(HippoStdNodeType.NT_FOLDER)
379                 || !sourceFolderNode.isNodeType(HippoTranslationNodeType.NT_TRANSLATED)) {
380             return null;
381         }
382 
383         Node translatedFolderNode = null;
384 
385         final String translationId = JcrUtils.getStringProperty(sourceFolderNode, HippoTranslationNodeType.ID, null);
386         final String statement = MessageFormat.format(TRANSLATED_FOLDER_QUERY, targetContentBaseNode.getPath(),
387                 translationId);
388         final Query query = targetContentBaseNode.getSession().getWorkspace().getQueryManager().createQuery(statement,
389                 Query.XPATH);
390 
391         final List<Node> translatedFolderNodes = new ArrayList<>();
392         final QueryResult result = query.execute();
393         Node node;
394 
395         for (NodeIterator nodeIt = result.getNodes(); nodeIt.hasNext();) {
396             node = nodeIt.nextNode();
397             if (node != null) {
398                 translatedFolderNodes.add(node);
399             }
400         }
401 
402         if (!translatedFolderNodes.isEmpty()) {
403             translatedFolderNode = translatedFolderNodes.get(0);
404 
405             if (translatedFolderNodes.size() > 1) {
406                 List<String> translatedFolderNodePaths = new ArrayList<>();
407                 for (Node folderNode : translatedFolderNodes) {
408                     translatedFolderNodePaths.add(folderNode.getPath());
409                 }
410                 log.warn("Multiple translated folder nodes found for translation ID, '{}': {}", translationId,
411                         translatedFolderNodePaths);
412             }
413         }
414 
415         return translatedFolderNode;
416     }

419      * Find translated document handle node under {@code targetContentBaseNode} for the {@code sourceDocumentHandleNode}.
420      * @param targetContentBaseNode target content base folder node
421      * @param sourceDocumentHandleNode source document handle node
422      * @return translated document handle node under {@code targetContentBaseNode} for the {@code sourceDocumentHandleNode}
423      * @throws RepositoryException if repository exception occurs

427         if (!sourceDocumentHandleNode.isNodeType(HippoNodeType.NT_HANDLE)
428                 || !sourceDocumentHandleNode.hasNode(sourceDocumentHandleNode.getName())) {
429             return null;
430         }
431 
432         final Node sourceDocumentVariantNode = sourceDocumentHandleNode.getNode(sourceDocumentHandleNode.getName());
433 
434         if (!sourceDocumentVariantNode.isNodeType(HippoTranslationNodeType.NT_TRANSLATED)) {
435             return null;
436         }
437 
438         final String translationId = JcrUtils.getStringProperty(sourceDocumentVariantNode, HippoTranslationNodeType.ID, null);
439 
440         final String statement = MessageFormat.format(TRANSLATED_DOCUMENT_HANDLE_QUERY, targetContentBaseNode.getPath(),
441                 translationId);

443                 Query.XPATH);
444

446         final QueryResult result = query.execute();
447         Node node;
448 
449         for (NodeIterator nodeIt = result.getNodes(); nodeIt.hasNext();) {
450             node = nodeIt.nextNode();
451             if (node != null) {
452                 translatedDocumentHandleNodes.add(node);

459             translatedDocumentHandleNode = translatedDocumentHandleNodes.get(0);

495     public static class FilterPresentComponentConfigurations implements Predicate<HstComponentConfiguration> {
496 
497         private final Set<String> filteredConfigurationUUIDs;
498         public FilterPresentComponentConfigurations(final HstComponentConfiguration sourceConfig, final HstSite sourceSite, final HstSite targetSite, final Session session) {
499             filteredConfigurationUUIDs =  new HashSet<>();
500             populateSkipSet(sourceConfig, sourceSite, targetSite, session, filteredConfigurationUUIDs);
501         }
502 
503         private void populateSkipSet(final HstComponentConfiguration sourceConfig,
504                                      final HstSite sourceSite,
505                                      final HstSite targetSite,
506                                      final Session session,
507                                      final Set<String> skipSet) {
508             try {
509                 final Node sourceNode = session.getNodeByIdentifier(sourceConfig.getCanonicalIdentifier());
510                 if (sourceNode.hasProperty(COMPONENT_PROPERTY_REFERECENCECOMPONENT)) {
511                     String reference = sourceNode.getProperty(COMPONENT_PROPERTY_REFERECENCECOMPONENT).getString();
512                     if (!StringUtils.isBlank(reference)) {
513                         final HstComponentConfiguration targetReference = targetSite.getComponentsConfiguration().getComponentConfiguration(reference);
514                         final HstComponentConfiguration sourceReference = sourceSite.getComponentsConfiguration().getComponentConfiguration(reference);
515                         if (targetReference != null) {
516                             log.debug("Skipping '{}' and descendants because targetSite '{}' already has a resolvable reference for '{}'",
517                                     sourceConfig, targetSite, reference);
518                             if (sourceReference != null) {
519                                 // sourceReference is never expected to be null, but just in case a null check
520                                 populateSelfAndDescending(sourceReference, skipSet);
521                             }
522                             // no need to check descendant configurations
523                             return;
524                         }
525                     }
526                 }
527             } catch (RepositoryException e) {
528                 throw new RuntimeRepositoryException(e);
529             }
530             for (HstComponentConfiguration child : sourceConfig.getChildren().values()) {
531                 populateSkipSet(child, sourceSite, targetSite, session, skipSet);
532             }
533         }
534 
535         private void populateSelfAndDescending(final HstComponentConfiguration current,
536                                                final Set<String> skipSet) {
537             skipSet.add(current.getCanonicalIdentifier());
538             for (HstComponentConfiguration child : current.getChildren().values()) {
539                 populateSelfAndDescending(child, skipSet);
540             }
541         }

546                 return false;

548             return true;

552     private void translateFolders(final Session session, final Node sourceBaseFolderNode,
553             final String sourceFolderRelPath, final Node targetBaseFolderNode, final String targetFolderRelPath,
554             final String targetTranslationLanguage) throws Exception {
555         String[] sourceFolderNodeNames = StringUtils.split(sourceFolderRelPath, "/");
556         String[] targetFolderNodeNames = StringUtils.split(targetFolderRelPath, "/");
557         String sourceFolderLocation = sourceBaseFolderNode.getPath();
558         String targetFolderLocation = targetBaseFolderNode.getPath();
559 
560         String sourceFolderNodeName;
561         String targetFolderNodeName;
562 
563         for (int i = 0; i < sourceFolderNodeNames.length; i++) {
564             sourceFolderNodeName = sourceFolderNodeNames[i];
565             targetFolderNodeName = (targetFolderNodeNames.length > i) ? targetFolderNodeNames[i] : sourceFolderNodeName;
566 
567             sourceFolderLocation += "/" + sourceFolderNodeName;
568 
569             if (!HippoFolderDocumentUtils.folderExists(session, sourceFolderLocation)) {
570                 throw new IllegalArgumentException("Source folder doesn't exist at '" + sourceFolderLocation + "'.");
571             }
572 
573             targetFolderLocation += "/" + targetFolderNodeName;
574 
575             if (!HippoFolderDocumentUtils.folderExists(session, targetFolderLocation)) {
576                 getDocumentManagementServiceClient().translateFolder(sourceFolderLocation, targetTranslationLanguage,
577                         targetFolderNodeName);
578             }
579         }

582     private static <T> List<T> popAllToList(final Stack<T> stack) {
583         if (stack == null) {
584             return Collections.emptyList();
585         }
586 
587         List<T> list = new LinkedList<>();
588         T object;
589 
590         while (!stack.empty()) {
591             object = stack.pop();
592             list.add(object);
593         }
594 
595         return list;
596     }
